-- SQL Schema for Menutech Menus Module
-- This module allows restaurants to manage their digital menus.

-- 1. Menus Table
CREATE TABLE IF NOT EXISTS public.menutech_menus (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    user_id UUID REFERENCES public.profiles(id) ON DELETE CASCADE,
    domain TEXT NOT NULL,
    cover_url TEXT,
    cover_type TEXT DEFAULT 'image', -- 'image' or 'video'
    slug TEXT UNIQUE,
    menu_style TEXT DEFAULT 'mode1',
    config JSONB DEFAULT '{
        "categories": [],
        "toppings": []
    }'::jsonb,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW(),
    UNIQUE(user_id)
);

-- Ensure slug and style columns exist if table was already created
DO $$
BEGIN
    IF NOT EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name='menutech_menus' AND column_name='slug') THEN
        ALTER TABLE public.menutech_menus ADD COLUMN slug TEXT UNIQUE;
    END IF;
    IF NOT EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name='menutech_menus' AND column_name='menu_style') THEN
        ALTER TABLE public.menutech_menus ADD COLUMN menu_style TEXT DEFAULT 'mode1';
    END IF;
END $$;

-- 2. Indexes for faster lookups
CREATE INDEX IF NOT EXISTS idx_menus_domain ON public.menutech_menus(domain);
CREATE INDEX IF NOT EXISTS idx_menus_slug ON public.menutech_menus(slug);
CREATE INDEX IF NOT EXISTS idx_menus_user_id ON public.menutech_menus(user_id);

-- 3. Enable Row Level Security
ALTER TABLE public.menutech_menus ENABLE ROW LEVEL SECURITY;

-- 4. RLS Policies

-- Public read access: Anyone can view a menu by its domain or slug
DO $$ BEGIN
    DROP POLICY IF EXISTS "Public read access for menus" ON public.menutech_menus;
    CREATE POLICY "Public read access for menus" ON public.menutech_menus FOR SELECT USING (true);
EXCEPTION WHEN others THEN NULL; END $$;

-- Management access: Owners can manage their own menu, admins/devs/CS can manage all
DO $$ BEGIN
    DROP POLICY IF EXISTS "Management access for menus" ON public.menutech_menus;
    CREATE POLICY "Management access for menus" ON public.menutech_menus FOR ALL USING (
        auth.uid() = user_id OR
        EXISTS (
            SELECT 1 FROM public.profiles
            WHERE profiles.id = auth.uid()
            AND (profiles.role = 'admin' OR profiles.role = 'developer' OR profiles.role = 'adminCS' OR profiles.role = 'CS')
        )
    );
EXCEPTION WHEN others THEN NULL; END $$;

-- 5. Trigger for updated_at
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = NOW();
    RETURN NEW;
END;
$$ language 'plpgsql';

DO $$ BEGIN
    CREATE TRIGGER update_menutech_menus_updated_at
    BEFORE UPDATE ON public.menutech_menus
    FOR EACH ROW
    EXECUTE PROCEDURE update_updated_at_column();
EXCEPTION WHEN others THEN NULL; END $$;

-- 6. Orders Table
CREATE TABLE IF NOT EXISTS public.menutech_orders (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    user_id UUID REFERENCES public.profiles(id) ON DELETE CASCADE,
    items JSONB DEFAULT '[]'::jsonb,
    total DECIMAL(10,2),
    status TEXT DEFAULT 'pending', -- 'pending', 'accepted', 'rejected'
    customer_name TEXT,
    customer_phone TEXT,
    customer_notes TEXT,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS idx_orders_user_id ON public.menutech_orders(user_id);
CREATE INDEX IF NOT EXISTS idx_orders_status ON public.menutech_orders(status);

ALTER TABLE public.menutech_orders ENABLE ROW LEVEL SECURITY;

DO $$ BEGIN
    DROP POLICY IF EXISTS "Public can create orders" ON public.menutech_orders;
    CREATE POLICY "Public can create orders" ON public.menutech_orders FOR INSERT WITH CHECK (true);
EXCEPTION WHEN others THEN NULL; END $$;

DO $$ BEGIN
    DROP POLICY IF EXISTS "Management access for orders" ON public.menutech_orders;
    CREATE POLICY "Management access for orders" ON public.menutech_orders FOR ALL USING (
        auth.uid() = user_id OR
        EXISTS (
            SELECT 1 FROM public.profiles
            WHERE profiles.id = auth.uid()
            AND (profiles.role = 'admin' OR profiles.role = 'developer' OR profiles.role = 'adminCS' OR profiles.role = 'CS')
        )
    );
EXCEPTION WHEN others THEN NULL; END $$;

DO $$ BEGIN
    CREATE TRIGGER update_menutech_orders_updated_at
    BEFORE UPDATE ON public.menutech_orders
    FOR EACH ROW
    EXECUTE PROCEDURE update_updated_at_column();
EXCEPTION WHEN others THEN NULL; END $$;
