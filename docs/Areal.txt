-- SQL para restaurar acceso y otorgar permisos a roles CS y ADMINCS
-- Ejecutar en el Editor SQL de Supabase

-- 1. Permitir LECTURA de perfiles a todos los usuarios autenticados (Sencillo y efectivo)
-- Esto restaura inmediatamente la visibilidad del nombre y rol del admin.
DROP POLICY IF EXISTS "Allow select for privileged roles" ON public.profiles;
CREATE POLICY "Allow select for authenticated" ON public.profiles
    FOR SELECT TO authenticated USING (true);

-- 2. Permitir ACTUALIZACIÓN de perfiles (Solo dueños y roles privilegiados)
DROP POLICY IF EXISTS "Allow update for privileged roles" ON public.profiles;
CREATE POLICY "Allow update for privileged roles" ON public.profiles
    FOR UPDATE TO authenticated
    USING (
        auth.uid() = id OR
        (SELECT role FROM public.profiles WHERE id = auth.uid()) IN ('admin', 'developer', 'CS', 'adminCS')
    );
