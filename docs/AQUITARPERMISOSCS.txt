-- SQL para otorgar permisos a roles CS y ADMINCS
-- Ejecutar en el Editor SQL de Supabase

-- 1. Actualizar política de perfiles para permitir lectura a roles CS
-- (Asegura que el sidebar pueda listar a todos los usuarios)
DROP POLICY IF EXISTS "Allow select for privileged roles" ON public.profiles;
CREATE POLICY "Allow select for privileged roles" ON public.profiles
    FOR SELECT USING (
        auth.uid() = id OR
        EXISTS (
            SELECT 1 FROM public.profiles
            WHERE profiles.id = auth.uid()
            AND UPPER(profiles.role) IN ('ADMIN', 'DEVELOPER', 'CS', 'ADMINCS')
        )
    );

-- 2. Actualizar política de Website Templates (menutech_web_templates)
DROP POLICY IF EXISTS "Template management access" ON public.menutech_web_templates;
CREATE POLICY "Template management access" ON public.menutech_web_templates
    FOR ALL USING (
        auth.uid() = user_id OR
        EXISTS (
            SELECT 1 FROM public.profiles
            WHERE profiles.id = auth.uid()
            AND UPPER(profiles.role) IN ('ADMIN', 'DEVELOPER', 'CS', 'ADMINCS')
        )
    );
